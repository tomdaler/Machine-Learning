STR_SAMPLE ="Dr. Strange loves pav bhaji. Hulk loves chaat of Delhi"


SPACY
=====

import spacy

nlp = spacy.load("en_core_web_sm")

doc = nlp(STR_SAMPLE)

for sentence in doc.sents:
    print(sentence)

    for word in sentence:
         print(word)
